SELECT ALL pokemons

SELECT * FROM pokemons;

SELECT especific COLUMNS

SELECT name,type FROM pokemons;

SELECT COLUMNS DISTINCTS

SELECT DISTINCT type FROM pokemons;

SELECT WITH CONDITION (WHERE)

SELECT name,type FROM pokemons WHERE type = 'Electric';

SELECT WITH CONDITION (OR WHERE)

SELECT name,type FROM pokemons WHERE type = 'Electric' OR type = 'Water';

SELECT WITH CONDITION (AND WHERE)

SELECT name, type  from pokemons where type='water' AND strenght>500

SELECT WITH CONDITION (>)

SELECT name, type  from pokemons where strenght>700

SELECT WITH CONDITION (<>)

SELECT name, type  from pokemons where type <> 'water';

SELECT WITH CONDITION (BETWEEN)

SELECT name, type, strenght  from pokemons where strenght BETWEEN 500 and 1000;

SELECT ORDER BY (ASC/DESC)

SELECT name, type, strenght  from pokemons ORDER BY name asc;

SELECT LIMIT

SELECT name, type, strenght  from pokemons ORDER BY name asc LIMIT 5;

SELECT LIMIT (with offset)

SELECT name, type, strenght  from pokemons ORDER BY name asc LIMIT 5 OFFSET 4;

SELECT OFFSET

SELECT * FROM pokemons OFFSET 2 LIMIT 3;

SELECT SEARCH (LIKE: Start with W)

SELECT name, type FROM pokemons WHERE name LIKE 'W%';

SELECT SEARCH (LIKE: End with e)

SELECT name, type FROM pokemons WHERE name LIKE '%e';

SELECT SEARCH (LIKE: Contains)

SELECT name, type FROM pokemons WHERE name LIKE '%ik%';

SELECT SEARCH (NOT LIKE: Contains)

SELECT name,type FROM pokemons WHERE name NOT LIKE '%on%';

SELECT SEARCH (LIKE: Complete)

SELECT name FROM pokemons WHERE name LIKE '_as%';

SELECT SEARCH (IN)

SELECT name, type, strenght FROM pokemons WHERE type IN ('water', 'Ghost');

SELECT CHALLENGES

List pokemons type water average speed 
SELECT AVG(speed) AS 'Velocidad de Acuaticos' FROM pokemons WHERE type = 'Water';
List all pokemons with total sum all skills 
SELECT name, (strenght + stamina + speed + accuracy) AS 'Total de habilidad' FROM pokemons;
Sum all skills by type 
SELECT type, SUM(strenght) AS 'Total de Fuerza' 
FROM pokemons 
GROUP BY type;
Count all pokemons type water 
SELECT type, COUNT(id) AS 'Cantidad de Acuaticos' 
FROM pokemons 
WHERE type = 'Water' 
GROUP BY type;

SELECT (ALIAS: Two Tables)

List all trainers with their pokemons 
SELECT T.name AS 'Entrenador', P.name AS 'Pokemon'
FROM trainers T, pokemons P
WHERE T.id = P.trainer_id;

SELECT (ALIAS: Two Tables with ORDER BY)

List all trainers with their pokemons ordered desc 
SELECT T.name AS 'Entrenador', P.name AS 'Pokemon' 
FROM trainers T, pokemons P 
WHERE T.id = P.trainer_id ORDER BY T.name desc,P.name DESC;

SELECT (ALIAS: Two Tables with AND OR & ORDER BY)

List all trainers with their pokemons type water or fire 
SELECT T.name AS 'Entrenador', P.name AS 'Pokemon', P.type 
FROM trainers T, pokemons P 
WHERE T.id = P.trainer_id AND (P.type = 'Water' OR P.type = 'Fire');

SELECT (ALIAS: Two Tables - COUNT)

List name trainer with number of pokemons 
SELECT T.name AS 'Entrenador', 
COUNT(P.id) AS 'Cantidad de Pokemones' 
FROM trainers T, pokemons P 
WHERE T.id = P.trainer_id GROUP BY T.name;

SELECT (ALIAS: Three Tables)

List all trainers with their pokemons & gyms 
SELECT G.name AS 'GYM', T.name AS 'Entrenador', P.name AS 'Pokemon'
FROM gyms G, trainers T, pokemons P 
WHERE G.id = T.gym_id AND T.id = P.trainer_id;

SELECT (INNER JOIN: Three Tables)

List all trainers with their pokemons & gyms 
SELECT G.name AS 'GYM', T.name AS 'Entrenador', P.name AS 'Pokemon' 
FROM gyms G 
INNER JOIN trainers T ON G.id = T.gym_id 
INNER JOIN pokemons P ON T.id = P.trainer_id;

SELECT (LEFT JOIN: Three Tables)

List all trainers with their pokemons & gyms 
SELECT G.name AS 'GYM', T.name AS 'Entrenador', P.name AS 'Pokemon' 
FROM gyms G LEFT JOIN trainers T ON G.id = T.gym_id 
LEFT JOIN pokemons P ON T.id = P.trainer_id;

SELECT (RIGHT JOIN: Three Tables)

List all trainers with their pokemons & gyms 
SELECT G.name AS 'GYM', T.name AS 'Entrenador', P.name AS 'Pokemon' 
FROM gyms G RIGHT JOIN trainers T ON G.id = T.gym_id 
RIGHT JOIN pokemons P ON T.id = P.trainer_id;

SELECT (UNION: Three Tables)

List all trainers, pokemons & gyms in a just one column 
SELECT name FROM gyms UNION SELECT name 
FROM trainers UNION SELECT name FROM pokemons;

CREATE VIEW

Create a view that contains all trainers with their pokemons & gyms 
CREATE VIEW Vista_Pokeadso AS SELECT G.name AS 'GYM', T.name AS 'Entrenador', P.name AS 'Pokemon' 
FROM gyms G 
JOIN trainers T ON G.id = T.gym_id 
JOIN pokemons P ON T.id = P.trainer_id;

SHOW VIEW

Show a view that contains all trainers with their pokemons & gyms 
SELECT * FROM Vista_Pokeadso;
